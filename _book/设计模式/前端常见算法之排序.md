#   前端常见排序


## 快速排序
```js
void qsort(int l,int r)
{
    if(l>=r) return ;
    int i = l,
        j = r;
    int tmp = A[l];
    while(i<j) {
        while(i<j && tmp <A[j]) j--; // 注意这里是-- 别忘了i<j
        swap(A[i],A[j]);
        while(i<j && tmp>=A[i]) i++;
        swap(A[i],A[j]);
    }
    A[i] = tmp;
    qsort(l,i);
    qsort(i+1,r);
}
```

## 归并排序
```js
void join(int l,int r)
{
    int i = l,k = l;
    int mid = (l+r)>>1;
    int j = mid+1;
    while(i<=mid && j<=r) {
        if(A[i] < A[j]) B[k++] = A[i++];
        else B[k++] = A[j++];
    }
    while(i<=mid) B[k++] = A[i++];
    while(j<=r) B[k++] = A[j++];
    for(int i=l;i<=r;i++) A[i] = B[i];
}

void gsort(int l,int r)
{
    if(l>=r) return ;
    int mid = (l+r)>>1;
    gsort(l,mid);
    gsort(mid+1,r);
    join(l,r);

}
```
